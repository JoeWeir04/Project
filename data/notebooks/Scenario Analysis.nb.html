library(ggplot2)
library(tidyverse)
library(likert)
library(ggstats)
library(dplyr)

Read CSV

raw_data <- read.csv("../processed/AllVisualisations_cleaned.csv", 
                     header = TRUE, stringsAsFactors = FALSE)

# Remove first two rows (metadata?) and reset rownames
data <- raw_data[-c(1,2), ]
rownames(data) <- NULL

head(data)
data$Q3_1 <- as.numeric(data$Q3_1)
data$Q3_2 <- as.numeric(data$Q3_2)
data$Q3_3 <- as.numeric(data$Q3_3)
data$Q3_4 <- as.numeric(data$Q3_4)
data$Q3_5 <- as.numeric(data$Q3_5)
data$Q3_6 <- as.numeric(data$Q3_6)
data$Q3_7 <- as.numeric(data$Q3_7)
data$Q3_8 <- as.numeric(data$Q3_8)

likert_levels <- c(
  "Strongly disagree",
  "Disagree",
  "Somewhat disagree",
  "Neutral",
  "Somewhat agree",
  "Agree",
  "Strongly agree"
)

scenarios <- data %>%
  select(Q3_1, Q3_2, Q3_3, Q3_4, Q3_5, Q3_6, Q3_7, Q3_8) %>%
  rename(
    `S1 Q1`  = Q3_1,
    `S1 Q2` = Q3_2,
    `S2 Q1`  = Q3_3,
    `S2 Q2` = Q3_4,
    `S3 Q1`  = Q3_5,
    `S3 Q2` = Q3_6,
    `S4 Q1`  = Q3_7,
    `S4 Q2` = Q3_8
  ) %>%
  mutate(across(everything(),
                ~ factor(.x, levels = 1:7, labels = likert_levels, ordered = TRUE)))
scenario_plot <- gglikert(
  scenarios,
  labels_size = 3,
  totals_include_center = TRUE
) +
  ggtitle("Scenario Responses")

ggsave(
  filename = "../graphs/Scenario_Responses.pdf",
  plot = scenario_plot,
  width = 9,
  height = 6,
  units = "in"
)

# Reshape Q2 data to long format per participant
distance_long <- data %>%
  select(Q4, Q2_1, Q2_2, Q2_3) %>%
  rename(
    PID    = Q4,
    Near   = Q2_1,
    Medium = Q2_2,
    Far    = Q2_3
  ) %>%
  pivot_longer(
    cols = c(Near, Medium, Far),
    names_to  = "Distance",
    values_to = "Visualisation"
  ) %>%
  mutate(
    Distance = factor(Distance, levels = c("Near", "Medium", "Far")),
    PID      = as.factor(PID)
  )

# Bump / slope chart: one line per participant
participant_switch_plot <- ggplot(distance_long,
                                  aes(x = Distance, y = Visualisation,
                                      group = PID, colour = PID)) +
  geom_line(linewidth = 0.8, alpha = 0.7) +
  geom_point(size = 3, alpha = 0.9) +
  labs(
    title   = "Per-Participant Visualisation Choice by Distance",
    subtitle = "Each line is one participant — crossing lines indicate switching behaviour",
    x       = "Distance from Sound Source",
    y       = "Preferred Visualisation",
    colour  = "Participant"
  ) +
  theme_minimal(base_size = 12) +
  theme(legend.position = "right")

print(participant_switch_plot)

ggsave(
  filename = "../graphs/Participant_Switch_by_Distance.pdf",
  plot     = participant_switch_plot,
  width    = 9,
  height   = 6,
  units    = "in"
)
